155      * Emulates one frame for the nes.
156      * @type void

159     frame:function nes_frame(){
160         //Clear the ppu buffer.
161         this.ppu.startFrame();
162         //Reset the cycle count to 0.
163         var cycles = 0;
164         //Start the frame loop.
165         FRAMELOOP:for(;;){
166             //Check if no cycles are to be halted.
167             if(this.cpu.cyclesToHalt === 0){
168                 //Execute a CPU instruction.
169                 cycles = this.cpu.emulate();
170                 //Set the cycles to the apu if active.
171                 if(this.apu.active){
172                     this.apu.clockFrameCounter(cycles);
173                 }
174                 //???
175                 cycles *= 3;
176             }
177             //Else check if the number of cycles to halt is less than or equal to 8.
178             else if(this.cpu.cyclesToHalt < 9){
179                 //???
180                 cycles = this.cpu.cyclesToHalt*3;
181                 //Set the cycles to halt to the apu if active.
182                 if(this.apu.active){
183                     this.apu.clockFrameCounter(this.cpu.cyclesToHalt);
184                 }
185                 //Set the cycles to halt to 0.
186                 this.cpu.cyclesToHalt = 0;
187             }
188             //Else the number of cycles to halt is greater than 8.
189             else{
190                 //Set the cycles to 24.
191                 cycles = 24;
192                 //Set the cycles to halt to the apu if active.
193                 if(this.apu.active){
194                     this.apu.clockFrameCounter(8);
195                 }
196                 //Remove 8 from the cycles to halt counter.
197                 this.cpu.cyclesToHalt -= 8;
198             }
199             //Loop for every cycle executed by the cpu.
200             for(;cycles>0;cycles--){
201                 //Check for a sprite 0 hit.
202                 if(this.ppu.curX === this.ppu.spr0HitX && this.ppu.f_spVisibility === 1 && this.ppu.scanline-21 === this.ppu.spr0HitY){
203                     //Set the sprite 0 hit flag.
204                     nes.cpu.mem[0x2002] |= 64;
205                 }
206                 //Check if the ppu is done rendering.
207                 if(this.ppu.requestEndFrame){
208                     //Decrement the non-maskable interrupt counter.
209                     this.ppu.nmiCounter--;
210                     //???
211                     if(this.ppu.nmiCounter === 0){
212                         //Reset the end of frame flag.
213                         this.ppu.requestEndFrame = false;
214                         //Start the vBlank period.
215                         this.ppu.endFrame();
216                         //Break the frame loop.
217                         break FRAMELOOP;
218                     }
219                 }
220                 //Increment the horizontal pixel counter.
221                 this.ppu.curX++;
222                 //Check if 341 pixels have been drawn, even though the scanlines are only 256 pixels wide.
223                 if(this.ppu.curX === 341){
224                     //If so reset the horizontal pixel counter to 0 and have the ppu end the scanline.
225                     this.ppu.curX = 0;
226                     this.ppu.endScanline();
227                 }
228             }
229         }
230 
231         //Calculate the fps.
232         var now = new Date().getTime();
233         var frameDifference = this.lastFrameTime-now;
234         this.fps = -1000/frameDifference;
235         this.lastFrameTime = now;

249     /**
250      * Loads the specified rom file, returns true if succesfully loaded.
251      * @type boolean
252      * @param {string} src The source of the rom file.
253      */
254 
255     loadRom:function nes_loadRom(src){
256         //Stop the emulator if running.
257         if(this.active){
258             this.stop();
259         }
260         //Create a new http request.
261         var request = new XMLHttpRequest();
262         request.open('GET',src,false);
263         //Set the mime type to binary.
264         request.overrideMimeType('text/plain; charset=x-user-defined');
265         //Send the request.
266         request.send(null);
267         //Cache the rom data.
268         var data = request.responseText;
269         //Check the rom validity.
270         if(data.indexOf("NES\x1a") !== -1){
271             //Create a blank object for the rom data.
272             this.rom = {};
273             //Reset the header.
274             this.rom.header = new Array(16);
275             //Load the header.
276             for(var i=0;i<16;i++){
277                 this.rom.header[i] = data.charCodeAt(i)&0xFF;
278             }
279             //Get the rom count.
280             this.rom.romCount = this.rom.header[4];
281             //Get the number of 4kb vrom banks, not 8kb.
282             this.rom.vromCount = this.rom.header[5]*2;
283             //Get the mirroring type.
284             this.rom.mirroring = (this.rom.header[6]&1);
285             //Check whether the rom has battery ram.
286             this.rom.hasBatteryRam = (this.rom.header[6]&2) !== 0;
287             //Get the trainer flag.
288             this.rom.trainer = (this.rom.header[6]&4) !== 0;
289             //Get the four screen flag.
290             this.rom.fourScreen = (this.rom.header[6]&8) !== 0;
291             //Get the mapper needed.
292             this.rom.mapperType = (this.rom.header[6]>>4)|(this.rom.header[7]&0xF0);
293             //Check whether any byte 8-15 is not a zero.
294             for(var i=8;i<16;i++){
295                 if(this.rom.header[i] !== 0){
296                     //Ignore byte 7.
297                     this.rom.mapperType &= 0xF;
298                 }
299             }
300             //Load PRG-ROM banks.
301             this.rom.rom = new Array(this.rom.romCount);
302             var offset = 16;
303             for(var i=0;i<this.rom.romCount;i++){
304                 this.rom.rom[i] = new Array(16384);
305                 for(var j=0;j<16384;j++){
306                     if(offset+j >= data.length){
307                         break;
308                     }
309                     this.rom.rom[i][j] = data.charCodeAt(offset+j)&0xFF;
310                 }
311                 offset += 16384;
312             }
313             //Reset the vrom and vrom tiles.
314             this.rom.vrom = new Array(this.rom.vromCount);
315             this.rom.vromTile = new Array(this.rom.vromCount);
316             //Loop through the vrom banks.
317             for(var i=0;i<this.rom.vromCount;i++){
318                 //Create the tiles.
319                 this.rom.vromTile[i] = new Array(256);
320                 for(var j=0;j<256;j++){
321                     this.rom.vromTile[i][j] = new Tile();
322                 }
323                 //Load the bank.
324                 this.rom.vrom[i] = new Array(4096);
325                 for(var j=0;j<4096;j++){
326                     if(offset+j >= data.length){
327                         break;
328                     }
329                     this.rom.vrom[i][j] = data.charCodeAt(offset+j)&0xFF;
330                 }
331                 offset += 4096;
332                 //Convert the bank to the tile.
333                 for(var j=0;j<4096;j++){
334                     if((j%16)<8){
335                         this.rom.vromTile[i][j>>4].setScanline(j%16,this.rom.vrom[i][j],this.rom.vrom[i][j+8]);
336                     }
337                     else{
338                         this.rom.vromTile[i][j>>4].setScanline((j%16)-8,this.rom.vrom[i][j-8],this.rom.vrom[i][j]);
339                     }
340                 }
341             }
342             //Reset the nes.
343             this.reset();
344             //Check the mapper needed.
345             if(typeof this.mappers[this.rom.mapperType] !== 'undefined'){
346                 //Save the rom's source.
347                 this.romSource = src;
348                 //Save the rom's filename, not including the extension.
349                 this.romName = src.substring(src.lastIndexOf('/')+1);
350                 this.romName = this.romName.substring(0,this.romName.lastIndexOf('.'));
351                 //Get the mapper.
352                 this.mmc = new this.mappers[this.rom.mapperType]();
353                 //Load the rom data.
354                 this.mmc.loadROM();
355                 //Set the ppu mirroring from the rom.
356                 if(this.rom.fourScreen){
357                     //Set fourscreen mirroring.
358                     this.ppu.setMirroring(2);
359                 }
360                 else if(this.rom.mirroring === 0){
361                     //Set horizontal mirroring.
362                     this.ppu.setMirroring(1);
363                 }
364                 else{
365                     //Set vertical mirroring.
366                     this.ppu.setMirroring(0);
367                 }
368                 //Rom was successfully loaded, return true.
369                 return true;
370             }
371             //Rom requires an unknown mapper, return false.
372             return false;
373         }
374         //Rom is not valid, return false.
375         return false;
376     },

383     loadBatteryRam:function nes_loadBatteryRam(){
384         //Check if the rom has batteryRAM.
385         if(nes.rom.hasBatteryRam){
386             //Check if the save file exists.
387             if(this.hasSave(this.romName)){
388                 //Get the save data.
389                 var data = this.getSave(this.romName);
390                 //Check the length of the data is valid.
391                 if(data.length === 8192){
392                     //Loop through the characters in the data.
393                     for(var i=0;i++;i<8192){
394                         //Set the corresponding address in memory as the character code minus 100.
395                         nes.cpu.mem[0x6000+i] = data.charCodeAt(i)-100;
396                     }
397                 }
398             }

408         //Check if the rom has battery ram.
409         if(nes.rom.hasBatteryRam){
410             //Create a string for storing the battery ram in the window.name property.
411             var batteryRam = '';
412             //Loop through the battery ram data.
413             for(var i=0x6000;i<0x8000;i++){
414                 //Add the battery ram data into the string by using characters to encode the numbers.
415                 //The extra value is added to keep odd characters from being used, as well as the semicolon which would skewer the data.
416                 batteryRam += String.fromCharCode(nes.cpu.mem[i]+100);
417             }
418             //Set the save data.
419             this.setSave(this.romName,batteryRam);
420         }

434     /**
435      * Sets the sent data to the name in the browser's window.name property, overwriting the old save data if it exists.
436      * @type void
437      * @param {string} name
438      * @param {string} value
439      */

441     setSave:function nes_setSave(name,value){
442         //Check if the name exists in the save data.
443         if(this.hasSave(name)){
444             //Get the save data.
445             var data = window.name.split(';');
446             //Remove the extra element added by split().
447             data.pop();
448             //Clear the window.name property.
449             window.name = '';
450             //Replace the old data associated with the name.
451             data[data.indexOf(name)+1] = value;
452             //Loop through the data elements and add them back into window.name property.
453             for(var i=0;i<data.length;i++){
454                 window.name += data[i]+';';
455             }
456         }
457         else{
458             //Else add the save data to the window.name property.
459             window.name += name+';'+value+';';
460         }

612             for(var i=0;i<61440;i++){
613                 //Check if the new and old colors are different.
614                 if(buffer[i] !== this.buffer[i]){
615                     //Cache the new color.

617                     //Set the red color component.

698         init:function nes_controller(){
699             //Define the key down event handler.
700             document.onkeydown = function document_onkeydown(event){
701                 //Loop through the key codes.
702                 for(var i=0;i<nes.controllers.keys1.length;i++){
703                     //Check the controller 1 code.
704                     if(event.keyCode === nes.controllers.keys1[i]){
705                         nes.controllers.state1[i] = 65;
706                         break;
707                     }
708                     //Else check the controller 2 code.
709                     else if(event.keyCode === nes.controllers.keys2[i]){
710                         nes.controllers.state2[i] = 65;
711                         break;
712                     }
713                 }
