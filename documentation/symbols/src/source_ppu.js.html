275     setMirroring:function ppu_setMirroring(mirroring){
276 
277         //???
278         this.triggerRendering();
279 
280         //Remove mirroring
281         this.vramMirrorTable = new Array(0x8000);
282         for(var i=0;i<0x8000;i++){
283             this.vramMirrorTable[i] = i;
284         }
285 
286         //Palette mirroring
287         this.defineMirrorRegion(0x3f20,0x3f00,0x20);
288         this.defineMirrorRegion(0x3f40,0x3f00,0x20);
289         this.defineMirrorRegion(0x3f80,0x3f00,0x20);
290         this.defineMirrorRegion(0x3fc0,0x3f00,0x20);
291 
292         //Additional mirroring
293         this.defineMirrorRegion(0x3000,0x2000,0xf00);
294         this.defineMirrorRegion(0x4000,0x0000,0x4000);
295 
296         //Horizontal mirroring
297         if(mirroring === 1){
298             this.ntable1[0] = 0;
299             this.ntable1[1] = 0;
300             this.ntable1[2] = 1;
301             this.ntable1[3] = 1;
302             this.defineMirrorRegion(0x2400,0x2000,0x400);
303             this.defineMirrorRegion(0x2c00,0x2800,0x400);
304         }
305 
306         //Vertical mirroring
307         else if(mirroring === 0){
308             this.ntable1[0] = 0;
309             this.ntable1[1] = 1;
310             this.ntable1[2] = 0;
311             this.ntable1[3] = 1;
312             this.defineMirrorRegion(0x2800,0x2000,0x400);
313             this.defineMirrorRegion(0x2c00,0x2400,0x400);
314         }
315 
316         //Single Screen mirroring
317         else if(mirroring === 3){
318             this.ntable1[0] = 0;
319             this.ntable1[1] = 0;
320             this.ntable1[2] = 0;
321             this.ntable1[3] = 0;
322             this.defineMirrorRegion(0x2400,0x2000,0x400);
323             this.defineMirrorRegion(0x2800,0x2000,0x400);
324             this.defineMirrorRegion(0x2c00,0x2000,0x400);
325         }
326 
327         //Single Screen mirroring 2
328         else if(mirroring === 4){
329             this.ntable1[0] = 1;
330             this.ntable1[1] = 1;
331             this.ntable1[2] = 1;
332             this.ntable1[3] = 1;
333             this.defineMirrorRegion(0x2400,0x2400,0x400);
334             this.defineMirrorRegion(0x2800,0x2400,0x400);
335             this.defineMirrorRegion(0x2c00,0x2400,0x400);
336         }
337 
338         //Assume Four-screen mirroring
339         else{
340             this.ntable1[0] = 0;
341             this.ntable1[1] = 1;
342             this.ntable1[2] = 2;
343             this.ntable1[3] = 3;
344         }

356     startFrame:function ppu_startFrame(){
357 
358         //Set the default bg color to black.
359         var bgColor = 0;
360 
361         //Check if monochrome.
362         if(this.f_dispType === 0){
363             //No, use first entry of image palette as bg color.
364             bgColor = this.imgPalette[0];
365         }
366         else{
367             //Yes, color emphasis determines the bg color.
368             switch(this.f_color){
369                 case 1:
370                     //Green
371                     bgColor = 0x00FF00;
372                     break;
373                 case 2:
374                     //Blue
375                     bgColor = 0xFF0000;
376                     break;
377                 case 4:
378                     //Red
379                     bgColor = 0x0000FF;
380                     break;
381             }
382         }
383 
384         //Loop through each pixel.
385         for(var i=0;i<61440;i++){
386 
387             //Set the bg color.
388             this.buffer[i] = bgColor;
389 
390             //???
391             this.pixRendered[i] = 65;
392 
393         }

398 
399         //Do the non-maskable interrupt.
400         nes.cpu.requestInterrupt(1);
401 
402         //Make sure everything is rendered.
403         if(this.lastRenderedScanline < 239){
404             this.renderFramePartially(this.lastRenderedScanline+1,240-this.lastRenderedScanline);
405         }

412         //           buffer[(this.sprY[0]<<8)+i] = 0xFF5555;
413         //       }
414         //       for(var i=0;i<240;i++){
415         //           buffer[(i<<8)+this.sprX[0]] = 0xFF5555;
416         //       }
417         //   }
418         //   //Hit position

420         //       for(var i=0;i<256;i++){
421         //           buffer[(this.spr0HitY<<8)+i] = 0x55FF55;
422         //       }
423         //       for(var i=0;i<240;i++){
424         //           buffer[(i<<8)+this.spr0HitX] = 0x55FF55;
425         //       }
426         //   }
427         //}
428 
429         //Check to clip the buffer to the tv size.
430         if(this.clipToTvSize){
431             //Clip the left and right pixels.
432             for(var y=0;y<240;y++){
433                 for(var x=0;x<8;x++){
434                     //Left

442                 for(var x=0;x<256;x++){

447                 }
448             }

452         else if(this.f_bgClipping === 0 || this.f_spClipping === 0){
453             //Clip left 8 pixels column.
454             for(var y=0;y<240;y++){
455                 for(var x=0;x<8;x++){
456                     this.buffer[(y<<8)+x] = 0;
457                 }
458             }
459         }
460 
461         //Write the buffer to the screen.
462         nes.screen.writeFrame(this.buffer);
463 
464         //Reset the last rendered scanline counter.
465         this.lastRenderedScanline = -1;

469     endScanline:function ppu_endScanline(){
470 
471         //Check if its a normal scanline.
472         if(this.scanline > 20 && this.scanline < 261){
473             //Check if the bg is visible.
474             if(this.f_bgVisibility === 1){
475                 //Update scroll.
476                 this.cntHT = this.regHT;
477                 this.cntH = this.regH;
478                 this.renderBgScanline(this.bgbuffer,this.scanline-20);
479                 //Check for sprite 0 hit on next scanline.
480                 if(!this.hitSpr0 && this.f_spVisibility === 1){
481                     if(this.checkSprite0(this.scanline-20)){
482                         this.hitSpr0 = true;
483                     }
484                 }
485                 //Clock mapper IRQ counter.
486                 nes.mmc.clockIrqCounter();
487             }
488             //Else check if the sprites are visible.
489             else if(this.f_spVisibility === 1){
490                 //Clock mapper IRQ counter.
491                 nes.mmc.clockIrqCounter();
492             }
493         }
494 
495         //Else check if its the dummy scanline.
496         else if(this.scanline === 19){
497             //May be variable length.
498             if(this.dummyCycleToggle){
499                 //Remove dead cycle at end of scanline, for next scanline.
500                 this.curX = 1;
501                 this.dummyCycleToggle = false;
502             }
503         }
504 
505         //Else check if its the first scanline.
506         else if(this.scanline === 20){
507             //Clear the vertical blank flag.
508             nes.cpu.mem[0x2002] &= 127;
509             //Clear the sprite 0 hit flags.
510             nes.cpu.mem[0x2002] &= 191;
511             this.hitSpr0 = false;
512             this.spr0HitX = -1;
513             this.spr0HitY = -1;
514             //Check if either the bg or sprites are visible.
515             if(this.f_bgVisibility === 1 || this.f_spVisibility === 1){
516                 //Update counters from registers.
517                 this.cntFV = this.regFV;
518                 this.cntV = this.regV;
519                 this.cntH = this.regH;
520                 this.cntVT = this.regVT;
521                 this.cntHT = this.regHT;
522                 //Check if the bg is visible.
523                 if(this.f_bgVisibility === 1){
524                     //Render dummy scanline.
525                     this.renderBgScanline(this.buffer,0);
526                 }
527                 //Check sprite 0 hit on the first scanline.
528                 this.checkSprite0(0);
529                 //Clock mapper IRQ Counter.
530                 nes.mmc.clockIrqCounter();
531             }
532         }
533 
534         //Else check if its the last(dead) scanline.
535         else if(this.scanline === 261){
536             //Set the vBlank flag.
537             nes.cpu.mem[0x2002] |= 128;
538             //Set the end frame flag.
539             this.requestEndFrame = true;
540             //???
541             this.nmiCounter = 9;
542             //Reset the scanline counter, will be incremented to 0.
543             this.scanline = -1;
544         }
545 
546         //Increment the scanline.
547         this.scanline++;
548 
549         //???
550         this.regsToAddress();
551         this.cntsToAddress();
552

555     writeRegister:function(address,value){
556         switch(address&7){
557 
558             //Write 0x2000, PPU Control Register
559             //Sets the Control Register.
560             case 0:
561                 //Set the value into the ram.
562                 nes.cpu.mem[0x2000] = value;
563                 //Update the internal buffers.
564                 nes.ppu.setControlRegister(value);
565                 break;
566 
567             //Write 0x2001, PPU Masking Register
568             //Sets the Masking Register.
569             case 1:
570                 //Set the value into the ram.
571                 nes.cpu.mem[0x2001] = value;
572                 //Update the internal buffers.
573                 nes.ppu.setMaskingRegister(value);
574                 break;
575 
576             //Write 0x2003, Sprite RAM Address
577             //Sets the address used when accessing sprite RAM.
578             case 3:
579                 //Set the address.
580                 this.sramAddress = value;
581                 break;
582 
583             //Write 0x2004, Sprite RAM
584             //Writes to the sprite RAM at the address set to 0x2003, increments the address afterwards.
585             case 4:
586                 //Write the value to the sprite ram.
587                 this.writeSpriteMem(this.sramAddress,value);
588                 //Increment the address.
589                 this.sramAddress++;
590                 //Knock it back down below 256.
591                 this.sramAddress %= 0x100;
592                 break;
593 
594             //Write 0x2005, Screen Scroll Offsets
595             //Sets the screen scrolling options?
596             case 5:
597                 //???
598                 this.triggerRendering();
599                 //Check if this is the first write.
600                 if(this.firstWrite){
601                     //Yes, set the horizontal scrolling.
602                     this.regHT = (value>>3)&31;
603                     this.regFH = value&7;
604                     this.firstWrite = false;
605                 }
606                 else{
607                     //No, set the vertical scrolling.
608                     this.regFV = value&7;
609                     this.regVT = (value>>3)&31;
610                     this.firstWrite = true;
611                 }
612                 break;
613 
614             //Write 0x2006, VRAM Address
615             //Sets the address used when accessing VRAM.
616             case 6:
617                 //Check if this is the first write.
618                 if(this.firstWrite){
619                     //Yes, set the high byte.
620                     this.regFV = (value>>4)&3;
621                     this.regV = (value>>3)&1;
622                     this.regH = (value>>2)&1;
623                     this.regVT = (this.regVT&7)|((value&3)<<3);
624                     this.firstWrite = false;
625                 }
626                 else{
627                     //No, set the low byte.
628                     this.triggerRendering();
629                     this.regVT = (this.regVT&24)|((value>>5)&7);
630                     this.regHT = value&31;
631                     this.cntFV = this.regFV;
632                     this.cntV = this.regV;
633                     this.cntH = this.regH;
634                     this.cntVT = this.regVT;
635                     this.cntHT = this.regHT;
636                     this.checkSprite0(this.scanline-20);
637                     this.firstWrite = true;
638                 }
639                 //Set the VRAM address from the counters just set.
640                 this.cntsToAddress();
641                 //Invoke mapper latch.
642                 if(this.vramAddress < 0x2000){
643                     nes.mmc.latchAccess(this.vramAddress);
644                 }
645                 break;
646 
647             //Write 0x2007, VRAM
648             case 7:
649                 //???
650                 this.triggerRendering();
651                 //???
652                 this.cntsToAddress();
653                 this.regsToAddress();
654                 //???
655                 if(this.vramAddress >= 0x2000){
656                     //Mirroring is used.
657                     this.mirroredWrite(this.vramAddress,value);
658                 }
659                 else{
660                     //Write normally.
661                     this.writeMem(this.vramAddress,value);
662                     //Invoke mapper latch.
663                     nes.mmc.latchAccess(this.vramAddress);
664                 }
665                 //Increment by either 1 or 32, depending on bit 2 of the control register.
666                 this.vramAddress += 1+this.f_addrInc*31;
667                 //???
668                 this.regsFromAddress();
669                 this.cntsFromAddress();
670                 break;
671 
672         }
673     },

722     //CPU Register $2007(R):
723     //Read from PPU memory. The address should be set first.
724     vramLoad:function(){
725         //???
726         this.cntsToAddress();
727         this.regsToAddress();
728         //If address is in range 0x0000-0x3EFF, return buffered values.
729         if(this.vramAddress <= 0x3EFF){
730             //Get the value from the previous read?
