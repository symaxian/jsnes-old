 55     write:function mmc0_write(address,value){
 56         //Check the address.
 57         if(address < 0x2000){
 58             //RAM
 59             nes.cpu.mem[address&0x7FF] = value;
 60         }
 61         else if(address < 0x4000){
 62             //PPU Registers
 63             nes.ppu.writeRegister(address,value);
 64         }
 65         else if(address < 0x4018){
 66             //IO & Sound Registers
 67             switch(address){
 68 
 69                 //Write 0x4014, Sprite Memory DMA Access
 70                 case 0x4014:
 71                     //Cache the base address.
 72                     var baseAddress = value*0x100;
 73                     //Cache the end address.
 74                     var endAddress = nes.ppu.sramAddress+256;
 75                     //Loop 256 times, for each byte to copy.
 76                     for(var i=nes.ppu.sramAddress;i<endAddress;i++){
 77                         //Write the byte from memory to the sprite ram.
 78                         nes.ppu.writeSpriteMem(i,nes.cpu.mem[baseAddress+i]);
 79                     }
 80                     //???
 81                     nes.cpu.haltCycles(513);
 82                     break;
 83 
 84                 //Write 0x4015, Sound Channel Switch, DMC Status
 85                 case 0x4015:
 86                     nes.apu.writeReg(address,value);
 87                     break;
 88 
 89                 //Write 0x4016, Joystick Strobe Reset
 90                 case 0x4016:
 91                     if(value === 0 && this.joypadLastWrite === 1){
 92                         this.joy1Strobe = 0;
 93                         this.joy2Strobe = 0;
 94                     }
 95                     this.joypadLastWrite = value;
 96                     break;
 97 
 98                 //Write 0x4000-0x4017, Sound Registers
 99                 default:
100                     nes.apu.writeReg(address,value);
101 
102             }
103         }
104         else{
105             //ROM
106             nes.cpu.mem[address] = value;
107         }
108     },

343 nes.mappers[1].prototype.reset = function mmc1_reset(){

363 nes.mappers[1].prototype.write = function mmc1_write(address,value){
364     //Writes below mmc1 registers are handled by mmc0.
365     if(address < 0x8000){
366         return nes.mappers.mmc0.prototype.write.apply(this,arguments);
367     }
368     //See what should be done with the written value.
369     if((value&128) !== 0){
370         //Reset buffering.
371         this.regBufferCounter = 0;
372         this.regBuffer = 0;
373         //Reset register.
374         if(this.getRegNumber(address) === 0){
375             this.prgSwitchingArea = 1;
376             this.prgSwitchingSize = 1;
377         }
378     }
379     else{
380         //Continue buffering.
381         this.regBuffer = (this.regBuffer&(0xFF-(1<<this.regBufferCounter)))|((value&1)<<this.regBufferCounter);
382         this.regBufferCounter++;
383         //???
384         if(this.regBufferCounter === 5){
385             //Use the buffered value.
386             this.setReg(this.getRegNumber(address),this.regBuffer);
387             //Reset buffer.
388             this.regBuffer = 0;
389             this.regBufferCounter = 0;
390         }
391     }
392 };

394 nes.mappers[1].prototype.setReg = function mmc1_setReg(reg,value){
395     switch(reg){
396 
397         //Register 0, PPU Mirroring, ROM loading flags.
398         case 0:
399             //Check if the ppu mirroring specified is different than the current.
400             if(value&3 !== this.mirroring){
401                 //Set the mirroring.
402                 this.mirroring = value&3;
403                 //Check for singlescreen mirroring.
404                 if((this.mirroring&2) === 0){
405                     nes.ppu.setMirroring(3);
406                 }
407                 //Else check for horizontal mirroring.
408                 else if((this.mirroring&1) !== 0){
409                     nes.ppu.setMirroring(1);
410                 }
411                 //Else set it as vertical mirroring.
412                 else{
413                     nes.ppu.setMirroring(0);
414                 }
415             }
416             //Set the PRG wwitching area.
417             this.prgSwitchingArea = (value>>2)&1;
418             //Set the PRG switching size.
419             this.prgSwitchingSize = (value>>3)&1;
420             //Set the VROM switching size.
421             this.vromSwitchingSize = (value>>4)&1;
422             break;
423 
424         //Register 1
425         case 1:
426             //Set the ROM selection register 0 flag.
427             this.romSelectionReg0 = (value>>4)&1;
428             //Check whether the cart has VROM.
429             if(nes.rom.vromCount > 0){
430                 //Select VROM bank at 0x0000.
431                 if(this.vromSwitchingSize === 0){
432                     //Swap 8kb VROM, the location depends on the romSelectionReg0 flag.
433                     this.load8kVromBank((value&0xF)+(this.romSelectionReg0*parseInt(nes.rom.vromCount/2,10)),0x0000);
434                 }
435                 else{
436                     //Swap 4kb VROM, the location depends on the romSelectionReg0 flag.
437                     this.load4kVromBank((value&0xF)+(this.romSelectionReg0*parseInt(nes.rom.vromCount/2,10)),0x0000);
438                 }
439             }
440             break;
441 
442         //Register 2
443         case 2:
444             //Set the ROM selection register 1 flag.
445             this.romSelectionReg1 = (value>>4)&1;
446             //Check whether the cart has VROM.
447             if(nes.rom.vromCount > 0){
448                 //Select VROM bank at 0x1000.
449                 if(this.vromSwitchingSize === 1){
450                     //Swap 4kB of VROM, the location depends on the romSelectionReg1 flag.
451                     this.load4kVromBank((value&0xF)+(this.romSelectionReg1*parseInt(nes.rom.vromCount/2,10)),0x1000);
452                 }
453             }
454             break;
455 
456         //Register 3, ROM bank select.
457         default:
458             //Initiate the base bank as 0.
459             var baseBank = 0;
460             //Check for a size 1024 kb cart.
461             if(nes.rom.romCount >= 32){
462                 if(this.vromSwitchingSize === 0 && this.romSelectionReg0 === 1){
463                     baseBank = 16;
464                 }
465                 else{
466                     baseBank = (this.romSelectionReg0|(this.romSelectionReg1<<1))<<3;
467                 }
468             }
469             //Else check for a size 512kb cart.
470             else if(nes.rom.romCount >= 16){
471                 //???
472                 if(this.romSelectionReg0 === 1){
473                     baseBank = 8;
474                 }
475             }
476             //Check for 32kb of rom.
477             if(this.prgSwitchingSize === 0){
478                 //Load the rom bank.
479                 this.load32kRomBank(baseBank+(value&0xF),0x8000);
480             }
481             //Else its 16kb or rom.
482             else{
483                 //Load the rom bank.
484                 this.load16kRomBank(baseBank*2+(value&0xF),0xC000-(this.prgSwitchingArea*0x4000));
485             }
486 
487     }
488 };

490 //Returns the register number from the address written to.
491 nes.mappers[1].prototype.getRegNumber = function(address){
492     //Return the value associated with the specified address.
493     if(address >= 0x8000 && address <= 0x9FFF){
494         return 0;
495     }
496     if(address >= 0xA000 && address <= 0xBFFF){
497         return 1;
498     }
499     if(address >= 0xC000 && address <= 0xDFFF){
500         return 2;
501     }
502     return 3;
503 };

557 nes.mappers[4] = function mmc4(){
558     //Initiate some variables.
559     this.command = null;
560     this.prgAddressSelect = null;
561     this.chrAddressSelect = null;
562     this.irqCounter = null;
563     this.irqLatchValue = null;
564     this.irqEnable = null;

570 nes.mappers[4].prototype.write = function mmc4_write(address,value){
571     //Writes below mmc4 registers are handled by mmc0.
572     if(address < 0x8000){
573         return nes.mappers.mmc0.prototype.write.apply(this,arguments);
574     }
575     //Switch through the address(registers).
576     switch(address){
577 
578         //Set Command/Address Register
579         case 0x8000:
580             //Set the command from the value.
581             this.command = value&7;
582             //Check to see if the program address specified in the value is different than the current one.
583             if((value>>6)&1 !== this.prgAddressSelect){
584                 //If so change it and flag it as having been changed.
585                 this.prgAddressSelect = (value>>6)&1;
586                 this.prgAddressChanged = true;
587             }
588             //Set the character address select flag from the value.
589             this.chrAddressSelect = (value>>7)&1;
590             break;
591 
592         //Execute Command
593         case 0x8001:
594             //Execute the command specified earlier with the specified value.
595             this.executeCommand(this.command,value);
596             break;
597 
598         //Set PPU Mirroring
599         case 0xA000:
600             //Set the ppu mirroring to vertical(0) or horizontal(1).
601             nes.ppu.setMirroring(value&1);
602             break;
603 
604         //SaveRAM Toggle
605         case 0xA001:
606             //Toggle the saveRAM, FIXME.
607             //nes.rom.setSaveState((value&1) !== 0);
608             break;
609 
610         //Interrupt Counter Register
611         case 0xC000:
612             //Set the interrupt counter to the value.
613             this.irqCounter = value;
614             //???
615             //nes.ppu.mapperIrqCounter = 0;
616             break;
617 
618         //Latch Interrupt Register
619         case 0xC001:
620             //Set the interrupt latch value.
621             this.irqLatchValue = value;
622             break;
623 
624         //Disable the Interrupt
625         case 0xE000:
626             //Set the interrupt enabled flag.
627             this.irqEnable = 0;
628             //???
629             //irqCounter = irqLatchValue;
630             break;
631 
632         //Enable the Interrupt
633         case 0xE001:
634             //Set the interrupt enabled flag.
635             this.irqEnable = 1;
636             break;
637 
638         //Unknown Register
639         default:
640             //Not a MMC4 register.
641             //The game has probably crashed, since it tries to write to rom at invalid locations.
642 
643     }
644 };

646 nes.mappers[4].prototype.executeCommand = function mmc4_executeCommand(cmd,arg){
647     switch(cmd){
648 
649         //Select 2 1kb VROM pages at 0x0000.
650         case 0:
651             //Load the 2 vrom banks specified by the character address flag.
652             this.load1kVromBank(arg,this.chrAddressSelect*0x1000);
653             this.load1kVromBank(arg+1,0x0400+(this.chrAddressSelect*0x1000));
654             break;
655 
656         //Select 2 1kb VROM pages at 0x0800.
657         case 1:
658             //Load the 2 vrom banks specified by the character address flag.
659             this.load1kVromBank(arg,0x0800+(this.chrAddressSelect*0x1000));
660             this.load1kVromBank(arg+1,0x0C00+(this.chrAddressSelect*0x1000));
661             break;
662 
663         //Select 1kb VROM Page at 0x1000.
664         case 2:
665             //Load the vrom bank specified by the character address select flag.
666             this.load1kVromBank(arg,this.chrAddressSelect*0x1000);
667             break;
668 
669         //Select 1k VROM Page at 0x1400.
670         case 3:
671             //Load the vrom bank specified by the character address select flag.
672             this.load1kVromBank(arg,0x1400-(this.chrAddressSelect*0x1000));
673             break;
674 
675         //Select 1k VROM Page at 0x1800.
676         case 4:
677             //Load the vrom bank specified by the character address select flag.
678             this.load1kVromBank(arg,0x1800-(this.chrAddressSelect*0x1000));
679             break;
680 
681         //Select 1k VROM Page at 0x1C00.
682         case 5:
683             //Load the vrom bank specified by the character address select flag.
684             this.load1kVromBank(arg,0x1C00-(this.chrAddressSelect*0x1000));
685             break;
686 
687         //Select ROM page 1.
688         case 6:
689             //Check if the rom bank address changed.
690             if(this.prgAddressChanged){
691                 //Load the hardwire bank specified by the program address select flag.
692                 this.load8kRomBank(((nes.rom.romCount-1)*2),0xC000-(this.prgAddressSelect*0x4000));
693                 //Reset the address changed flag.
694                 this.prgAddressChanged = false;
695             }
696             //Select first switchable ROM page.
697             this.load8kRomBank(arg,0x8000+(this.prgAddressSelect*0x4000));
698             break;
699 
700         //Select ROM page 2.
701         case 7:
702             //Select second switchable ROM page.
703             this.load8kRomBank(arg,0xA000);
704             //Check the the rom bank address changed.
705             if(this.prgAddressChanged){
706                 //Load the hardwired bank specified by the programm address select flag.
707                 this.load8kRomBank(((nes.rom.romCount-1)*2),0xC000-(this.prgAddressSelect*0x4000));
708                 //Reset the address changed flag.
709                 this.prgAddressChanged = false;
710             }
711             break;
712 
713     }
