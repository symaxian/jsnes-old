180     readReg:function nes_apu_readReg(){
181 
182         //Construct the value that would normally be held in 0x4015 from the apu flags.
183         var temp = (this.dmc.getIrqStatus()<<7)|(((this.frameIrqActive && this.frameIrqEnabled)?1:0)<<6)|(this.getChannelLengthStatus(this.dmc)<<4)|(this.getChannelLengthStatus(this.noise)<<3)|(this.getChannelLengthStatus(this.triangle)<<2)|(this.getChannelLengthStatus(this.square2)<<1)|this.getChannelLengthStatus(this.square1);
184 
185         //???
186         this.frameIrqActive = false;
187         this.dmc.irqGenerated = false;
188 
189         //Return the previously constructed byte.
190         return temp&0xFFFF;

194     writeReg:function nes_apu_writeReg(address,value){
195 
196         //Square Wave 1 Control
197         if(address >= 0x4000 && address < 0x4004){
198             this.square1.writeReg(address,value);
199         }
200 
201         //Square Wave 2 Control
202         else if(address >= 0x4004 && address < 0x4008){
203             this.square2.writeReg(address,value);
204         }
205 
206         //Triangle Control
207         else if(address >= 0x4008 && address < 0x400C){
208             this.triangle.writeReg(address,value);
209         }
210 
211         //Noise Control
212         else if(address >= 0x400C && address <= 0x400F){
213             this.noise.writeReg(address,value);
214         }
215 
216         //DMC Play mode & DMA frequency
217         else if(address === 0x4010){
218             this.dmc.writeReg(address,value);
219         }
220 
221         //DMC Delta Counter
222         else if(address === 0x4011){
223             this.dmc.writeReg(address,value);
224         }
225 
226         //DMC Play code starting address
227         else if(address === 0x4012){
228             this.dmc.writeReg(address,value);
229         }
230 
231         //DMC Play code length
232         else if(address === 0x4013){
233             this.dmc.writeReg(address,value);
234         }
235 
236         //Channel enable.
237         else if(address === 0x4015){
238             //???
239             this.updateChannelEnable(value);
240             //???
241             if(value !== 0 && this.initCounter > 0){
242                 //Start hardware initialization.
243                 this.initiatingHardware = true;
244             }
245             //DMC/IRQ Status
246             this.dmc.writeReg(address,value);
247         }
248 
249         //Frame counter control.
250         else if(address === 0x4017){
251             //???
252             this.countSequence = (value>>7)&1;
253             this.masterFrameCounter = 0;
254             this.frameIrqActive = false;
255             //???
256             this.frameIrqEnabled = (((value>>6)&0x1) === 0);
257             //???
258             if(this.countSequence === 0){
259                 //NTSC
260                 this.frameIrqCounterMax = 4;
261                 this.derivedFrameCounter = 4;
262             }
263             else{
264                 //PAL
265                 this.frameIrqCounterMax = 5;
266                 this.derivedFrameCounter = 0;
267                 this.frameCounterTick();
268             }
269         }

283     //Clocks the frame counter.
284     //It should be clocked at twice the cpu speed, so the cycles will be divided by 2 for those counters that are clocked at cpu speed.
285     clockFrameCounter:function nes_apu_clockFrameCounter(nCycles){
286 
287         if(this.initCounter > 0 && this.initiatingHardware){
288             this.initCounter -= nCycles;
289             if(this.initCounter <= 0){
290                 this.initiatingHardware = false;
291             }
292             return;
293         }
294 
295         //Don't process ticks beyond next sampling.
296         nCycles += this.extraCycles;
297         var maxCycles = this.sampleTimerMax-this.sampleTimer;
298         if((nCycles<<10) > maxCycles){
299             this.extraCycles = ((nCycles<<10)-maxCycles)>>10;
300             nCycles -= this.extraCycles;
301         }
302         else{
303             this.extraCycles = 0;
304         }
305 
306         //Clock the dm channel.
307         if(this.dmc.isEnabled){
308             this.dmc.shiftCounter -= (nCycles<<3);
309             while(this.dmc.shiftCounter <= 0 && this.dmc.dmaFrequency > 0){
310                 this.dmc.shiftCounter += this.dmc.dmaFrequency;
311             }
312         }
313 
314         //Clock the triangle channel.
315         if(this.triangle.progTimerMax > 0){
316             this.triangle.progTimerCount -= nCycles;
317             while(this.triangle.progTimerCount <= 0){
318                 this.triangle.progTimerCount += this.triangle.progTimerMax+1;
319                 if(this.triangle.linearCounter>0 && this.triangle.lengthCounter>0){
320                     this.triangle.triangleCounter++;
321                     this.triangle.triangleCounter &= 0x1F;
322                     if(this.triangle.isEnabled){
323                         if(this.triangle.triangleCounter>=0x10){
324                             //Normal value.
325                             this.triangle.sampleValue = (this.triangle.triangleCounter&0xF);
326                         }
327                         else{
328                             //Inverted value.
329                             this.triangle.sampleValue = (0xF - (this.triangle.triangleCounter&0xF));
330                         }
331                         this.triangle.sampleValue <<= 4;
332                     }
333                 }
334             }
335         }
336 
337         //Clock the square1 channel.
338         this.square1.progTimerCount -= nCycles;
339         if(this.square1.progTimerCount <= 0){
340             this.square1.progTimerCount += (this.square1.progTimerMax+1)<<1;
341             this.square1.squareCounter++;
342             this.square1.squareCounter &= 0x7;
343             this.square1.updateSampleValue();
344         }
345 
346         //Clock the square2 channel.
347         this.square2.progTimerCount -= nCycles;
348         if(this.square2.progTimerCount <= 0){
349             this.square2.progTimerCount += (this.square2.progTimerMax+1)<<1;
350             this.square2.squareCounter++;
351             this.square2.squareCounter &= 0x7;
352             this.square2.updateSampleValue();
353         }
354 
355         //Clock the noise channel.
356         var acc_c = nCycles;
357         if(this.noise.progTimerCount-acc_c > 0){
358             //Do all cycles at once.
359             this.noise.progTimerCount -= acc_c;
360             this.noise.accCount += acc_c;
361             this.noise.accValue += acc_c*this.noise.sampleValue;
362         }
363         else{
364             //Slow-step.
365             while((acc_c--) > 0){
366                 if(--this.noise.progTimerCount <= 0 && this.noise.progTimerMax > 0){
367                     //Update this.noise shift register.
368                     this.noise.shiftReg <<= 1;
369                     this.noise.tmp = (((this.noise.shiftReg<<(this.noise.randomMode === 0?1:6))^this.noise.shiftReg)&0x8000);
370                     if(this.noise.tmp !== 0){
371                         //Sample value must be 0.
372                         this.noise.shiftReg |= 0x01;
373                         this.noise.randomBit = 0;
374                         this.noise.sampleValue = 0;
375                     }
376                     else{
377                         //Find sample value.
378                         this.noise.randomBit = 1;
379                         if(this.noise.isEnabled && this.noise.lengthCounter>0){
380                             this.noise.sampleValue = this.noise.masterVolume;
381                         }
382                         else{
383                             this.noise.sampleValue = 0;
384                         }
385                     }
386                     this.noise.progTimerCount += this.noise.progTimerMax;
387                 }
388                 this.noise.accValue += this.noise.sampleValue;
389                 this.noise.accCount++;
390             }
391         }
392 
393         //Frame interrupt handling.
394         if(this.frameIrqEnabled && this.frameIrqActive){
395             nes.cpu.requestIrq(0);
396         }
397 
398         //Clock frame counter at double CPU speed.
399         this.masterFrameCounter += (nCycles<<1);
400         if(this.masterFrameCounter >= this.frameTime){
401             //240Hz tick.
402             this.masterFrameCounter -= this.frameTime;
403             this.frameCounterTick();
404         }
405 
406         //Accumulate sample values.
407 
408         //Special treatment for triangle channel, need to interpolate.
409         if(this.triangle.sampleCondition){
410             this.triValue = Math.min(parseInt((this.triangle.progTimerCount<<4)/(this.triangle.progTimerMax+1),10),16);
411             if(this.triangle.triangleCounter >= 16){
412                 this.triValue = 16-this.triValue;
413             }
414             //Add non-interpolated sample value.
415             this.triValue += this.triangle.sampleValue;
416         }
417 
418         //Now sample normally.
419         if(nCycles === 2){
420             this.smpTriangle += this.triValue<<1;
421             this.smpDmc += this.dmc.sample<<1;
422             this.smpSquare1 += this.square1.sampleValue<<1;
423             this.smpSquare2 += this.square2.sampleValue<<1;
424             this.accCount += 2;
425         }
426         else if(nCycles === 4){
427             this.smpTriangle += this.triValue<<2;
428             this.smpDmc += this.dmc.sample<<2;
429             this.smpSquare1 += this.square1.sampleValue<<2;
430             this.smpSquare2 += this.square2.sampleValue<<2;
431             this.accCount += 4;
432         }
433         else{
434             this.smpTriangle += nCycles*this.triValue;
435             this.smpDmc += nCycles*this.dmc.sample;
436             this.smpSquare1 += nCycles*this.square1.sampleValue;
437             this.smpSquare2 += nCycles*this.square2.sampleValue;
438             this.accCount += nCycles;
439         }
440 
441         //Clock sample timer.
442         this.sampleTimer += nCycles<<10;
443         if(this.sampleTimer >= this.sampleTimerMax){
444             //Sample channels.
445             this.sample();
446             this.sampleTimer -= this.sampleTimerMax;
447         }
448 
449     },

451     frameCounterTick:function nes_apu_frameCounterTick(){
452 
453         this.derivedFrameCounter++;
454         if(this.derivedFrameCounter >= this.frameIrqCounterMax){
455             this.derivedFrameCounter = 0;
456         }
457 
458         if(this.derivedFrameCounter === 1 || this.derivedFrameCounter === 3){
459             //Clock length & sweep.
460             this.triangle.clockLengthCounter();
461             this.square1.clockLengthCounter();
462             this.square2.clockLengthCounter();
463             this.noise.clockLengthCounter();
464             this.square1.clockSweep();
465             this.square2.clockSweep();
466         }
467 
468         if(this.derivedFrameCounter >= 0 && this.derivedFrameCounter < 4){
469             //Clock linear & decay.
470             this.square1.clockEnvDecay();
471             this.square2.clockEnvDecay();
472             this.noise.clockEnvDecay();
473             this.triangle.clockLinearCounter();
474         }
475 
476         if(this.derivedFrameCounter === 3 && this.countSequence === 0){
477             //Enable IRQ.
478             this.frameIrqActive = true;
479         }
480 
481         //End of 240Hz tick.
482 
483     },

485     //Samples the channels, mixes the output together, writes to buffer and (if enabled) file.
486     sample:function nes_apu_sample(){
487 
488         if(this.accCount > 0){
489             //Square Wave 1
490             this.smpSquare1 <<= 4;
491             this.smpSquare1 = parseInt(this.smpSquare1/this.accCount,10);
492             //Square Wave 2
493             this.smpSquare2 <<= 4;
494             this.smpSquare2 = parseInt(this.smpSquare2/this.accCount,10);
495             //Triangle Wave
496             this.smpTriangle = parseInt(this.smpTriangle/this.accCount,10);
497             //DMC
498             this.smpDmc <<= 4;
499             this.smpDmc = parseInt(this.smpDmc / this.accCount,10);
500             //???
501             this.accCount = 0;
502         }
503         else{
504             //Square Wave 1
505             this.smpSquare1 = this.square1.sampleValue<<4;
506             //Square Wave 2
507             this.smpSquare2 = this.square2.sampleValue<<4;
508             //Triangle Wave
509             this.smpTriangle = this.triangle.sampleValue;
510             //DMC
511             this.smpDmc = this.dmc.sample<<4;
512         }
513 
514         var smpNoise = parseInt((this.noise.accValue<<4)/this.noise.accCount,10);
515 
516         this.noise.accValue = smpNoise>>4;
517         this.noise.accCount = 1;
518     
519         //Left sound channel.
520         var sq_index  = (this.smpSquare1*this.stereoPosLSquare1+this.smpSquare2*this.stereoPosLSquare2)>>8;
521         var tnd_index = (3*this.smpTriangle*this.stereoPosLTriangle+(smpNoise<<1)*this.stereoPosLNoise+this.smpDmc*this.stereoPosLDMC)>>8;
522         if(sq_index >= this.square_table.length){
523             sq_index  = this.square_table.length-1;
524         }
525         if(tnd_index >= this.tnd_table.length){
526             tnd_index = this.tnd_table.length-1;
527         }
528         var sampleValueL = this.square_table[sq_index]+this.tnd_table[tnd_index]-this.dcValue;
529 
530         //Right sound channel.
531         sq_index = (this.smpSquare1*this.stereoPosRSquare1+this.smpSquare2*this.stereoPosRSquare2)>>8;
532         tnd_index = (3*this.smpTriangle*this.stereoPosRTriangle+(smpNoise<<1)*this.stereoPosRNoise+this.smpDmc*this.stereoPosRDMC)>>8;
533         if(sq_index >= this.square_table.length){
534             sq_index = this.square_table.length-1;
535         }
536         if(tnd_index >= this.tnd_table.length){
537             tnd_index = this.tnd_table.length-1;
538         }
539         var sampleValueR = this.square_table[sq_index]+this.tnd_table[tnd_index]-this.dcValue;
540 
541         //Remove DC from left channel.
542         var smpDiffL = sampleValueL-this.prevSampleL;
543         this.prevSampleL += smpDiffL;
544         this.smpAccumL += smpDiffL-(this.smpAccumL>>10);
545         sampleValueL = this.smpAccumL;
546 
547         //Remove DC from right channel.
548         var smpDiffR = sampleValueR-this.prevSampleR;
549         this.prevSampleR += smpDiffR;
550         this.smpAccumR += smpDiffR-(this.smpAccumR>>10);
551         sampleValueR = this.smpAccumR;
552 
553         //Check that the sample values are not greater than or less than the maximum and minimum sample values.
554         if(sampleValueL > this.maxSample){
555             this.maxSample = sampleValueL;
556         }
557         if(sampleValueL < this.minSample){
558             this.minSample = sampleValueL;
559         }
560 
561         //Add the left and right sample values to the buffer, incrementing the index as well.
562         this.sampleBuffer[this.bufferIndex++] = sampleValueL;
563         this.sampleBuffer[this.bufferIndex++] = sampleValueR;
564 
565         //Send the buffer to the speakers if its full(the index is at the end).
566         if(this.bufferIndex === this.sampleBuffer.length){
567             //Write the samples to the audio wrapper.
568             nes.dynamicAudio.writeInt(this.sampleBuffer);
569             //Reset the buffer and buffer index.
570             this.sampleBuffer = new Array(16384);
571             this.bufferIndex = 0;
572         }
573 
574         //Reset sampled values.
575         this.smpSquare1 = 0;
576         this.smpSquare2 = 0;
577         this.smpTriangle = 0;
578         this.smpDmc = 0;
579 
580     },

611     updateStereoPos:function nes_apu_updateSteroPos(){
612 
613         //Update left.
614         this.stereoPosLSquare1 = (this.panning[0]*this.masterVolume)>>8;
615         this.stereoPosLSquare2 = (this.panning[1]*this.masterVolume)>>8;
616         this.stereoPosLTriangle = (this.panning[2]*this.masterVolume)>>8;
617         this.stereoPosLNoise = (this.panning[3]*this.masterVolume)>>8;
618         this.stereoPosLDMC = (this.panning[4]*this.masterVolume)>>8;
619 
620         //Update right.
621         this.stereoPosRSquare1 = this.masterVolume-this.stereoPosLSquare1;
622         this.stereoPosRSquare2 = this.masterVolume-this.stereoPosLSquare2;
623         this.stereoPosRTriangle = this.masterVolume-this.stereoPosLTriangle;
624         this.stereoPosRNoise = this.masterVolume-this.stereoPosLNoise;
625         this.stereoPosRDMC = this.masterVolume-this.stereoPosLDMC;

629     initDACtables:function nes_apu_initDACtables(){
630 
631         var max_sqr = 0;
632         this.square_table = new Array(512);
633         for(var i=0;i<512;i++){
634             value = 95.52/(8128/(i/16)+100);
635             value *= 0.98411;
636             value *= 50000;
637             var ival = parseInt(value,10);
638             this.square_table[i] = ival;
639             if(ival > max_sqr){
640                 max_sqr = ival;
641             }
642         }
643 
644         var max_tnd = 0;
645         this.tnd_table = new Array(3264);
646         for(var i=0;i<3264;i++){
647             var value = 163.67/(24329/(i/16)+100);
648             value *= 0.98411;
649             value *= 50000;
650             ival = parseInt(value,10);
651             this.tnd_table[i] = ival;
652             if(ival > max_tnd){
653                 max_tnd = ival;
654             }
655         }
656 
657         this.dacRange = max_sqr+max_tnd;
658         this.dcValue = this.dacRange/2;
659 
660     },
